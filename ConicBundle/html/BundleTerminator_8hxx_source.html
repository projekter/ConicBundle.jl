130  virtual int check_termination(BundleTerminatorData* sb)
131  {
132  terminated=0;
133  if (clockp)
134  terminated|=2*(clockp->time()>=timelimit);
135  if (recomplimit>=0)
136  terminated|=4*(recomplimit<=sb->get_sumrecomp());
137  if (qpfailslimit>=0)
138  terminated|=8*(qpfailslimit<=sb->get_sumqpfails());
139  if (modelfailslimit>=0)
140  terminated|=16*(modelfailslimit<=sb->get_summodelfails());
141  if (augvalfailslimit>=0)
142  terminated|=32*(augvalfailslimit<=sb->get_sumaugvalfails());
143  if (objevallimit>=0)
144  terminated|=64*(objevallimit<=sb->get_cntobjeval());
145  if (oraclefailslimit>=0)
146  terminated|=128*(oraclefailslimit<=sb->get_sumoraclefails());
147  if ((sb->get_suminnerit()<10)&&(sb->get_aggr_dnormsqr()>0.1)) return terminated;
148  if ((aggregate_dnormsqr>0)&&(sb->get_aggr_dnormsqr()>aggregate_dnormsqr)) return terminated;
149  CH_Matrix_Classes::Real center_objval=sb->get_center_objval();
150  terminated|=(center_objval-sb->get_modelval()<=
151  termeps*(CH_Matrix_Classes::abs(center_objval)+1.)*sb->get_term_corr());
152  return terminated;
153  }
